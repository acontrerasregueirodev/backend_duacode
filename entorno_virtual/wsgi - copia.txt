import zipfile
import os

# Ruta del archivo ZIP
zip_file_path = '/home/belami/proyecto_duacode.zip'  # Cambia esta ruta al lugar donde está tu ZIP
# Nombre de la carpeta de extracción
extract_to = 'proyecto_duacode'

# Crear la carpeta de extracción si no existe
os.makedirs(extract_to, exist_ok=True)

# Descomprimir el archivo
with zipfile.ZipFile(zip_file_path, 'r') as zip_ref:
    zip_ref.extractall(extract_to)

print(f'Archivos descomprimidos en: {extract_to}')
